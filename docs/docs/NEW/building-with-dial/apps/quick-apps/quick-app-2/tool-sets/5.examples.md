# Quick App 2.0 examples

Copy-pasteable Quick App 2.0 configurations for common patterns. Each example shows a complete `applicationProperties` object ready to use with the [DIAL Core API](https://dialx.ai/dial_api#tag/Applications/operation/saveCustomApplication).

For field-level documentation, see the [configuration reference](4.reference.md).

---

## 1. Simple assistant (model + instructions only)

A Quick App with no tools — pure instruction-following.

```json
{
  "applicationTypeSchemaId": "https://mydial.epam.com/custom_application_schemas/quickapps2",
  "applicationProperties": {
    "orchestrator": {
      "deployment": {
        "name": "gpt-4o-mini",
        "parameters": {
          "temperature": 0.7
        }
      },
      "system_prompt": {
        "type": "custom",
        "variables": {},
        "content": "You are a helpful customer support assistant for Acme Corp. Be concise, friendly, and accurate. If you don't know something, say so clearly."
      }
    },
    "contexts": [],
    "tool_sets": [],
    "starters": [
      "How do I reset my password?",
      "Where can I find my invoice?",
      "How do I contact support?"
    ]
  }
}
```

## 2. Weather assistant (REST API tools, no auth)

Calls the free Open-Meteo API for current weather. Demonstrates two chained REST tools: geocoding and weather lookup.

```json
{
  "applicationTypeSchemaId": "https://mydial.epam.com/custom_application_schemas/quickapps2",
  "applicationProperties": {
    "orchestrator": {
      "deployment": {
        "name": "gpt-4o-mini"
      },
      "system_prompt": {
        "type": "custom",
        "variables": {},
        "content": "You are a weather assistant. When users ask about weather, first call geo_code to get coordinates for the location, then call weather to get the current temperature. Present results in a friendly, conversational way."
      }
    },
    "contexts": [],
    "tool_sets": [
      {
        "type": "rest-api",
        "name": "location-tools",
        "authorization": null,
        "tools": [
          {
            "type": "restapi-tool",
            "rest_api_method_info": {
              "method_url": "https://geocoding-api.open-meteo.com/v1/search",
              "method_type": "get"
            },
            "open_ai_tool": {
              "type": "function",
              "function": {
                "name": "geo_code",
                "description": "Get latitude and longitude for a city or location name.",
                "parameters": {
                  "type": "object",
                  "properties": {
                    "name": {
                      "type": "string",
                      "description": "City or location name to look up.",
                      "parameter_info": { "type": "query", "key": "name" }
                    },
                    "count": {
                      "type": null,
                      "const": "1",
                      "parameter_info": { "type": "query", "key": "count" }
                    }
                  },
                  "required": ["name", "count"]
                }
              }
            }
          },
          {
            "type": "restapi-tool",
            "rest_api_method_info": {
              "method_url": "https://api.open-meteo.com/v1/forecast",
              "method_type": "get"
            },
            "open_ai_tool": {
              "type": "function",
              "function": {
                "name": "weather",
                "description": "Get current temperature for geographic coordinates.",
                "parameters": {
                  "type": "object",
                  "properties": {
                    "latitude": {
                      "type": "number",
                      "description": "Latitude of the location.",
                      "parameter_info": { "type": "query", "key": "latitude" }
                    },
                    "longitude": {
                      "type": "number",
                      "description": "Longitude of the location.",
                      "parameter_info": { "type": "query", "key": "longitude" }
                    },
                    "current": {
                      "type": null,
                      "const": "temperature_2m,wind_speed_10m",
                      "parameter_info": { "type": "query", "key": "current" }
                    }
                  },
                  "required": ["latitude", "longitude", "current"]
                }
              }
            }
          }
        ]
      }
    ],
    "starters": [
      "What is the weather in Paris?",
      "Is it warm in Tokyo today?",
      "How cold is it in New York?"
    ]
  }
}
```

## 3. External MCP server (streamable HTTP, bearer auth)

Connects to an external MCP server over `streamable_http` with bearer token authentication. Only exposes specific tools from the server via `allowed_tools`.

```json
{
  "applicationTypeSchemaId": "https://mydial.epam.com/custom_application_schemas/quickapps2",
  "applicationProperties": {
    "orchestrator": {
      "deployment": {
        "name": "gpt-4o"
      },
      "system_prompt": {
        "type": "custom",
        "variables": {},
        "content": "You are a research assistant with access to a web search tool. When users ask questions that require current information, use the web_search tool to find relevant results before answering."
      }
    },
    "contexts": [],
    "tool_sets": [
      {
        "type": "mcp",
        "name": "web-search-mcp",
        "mcp_server_info": {
          "url": "https://mcp.example.com/search",
          "protocol": "streamable_http",
          "authorization": {
            "type": "bearer",
            "token": "your-bearer-token-here"
          }
        },
        "allowed_tools": ["web_search", "fetch_page"],
        "fallback_configuration": {
          "strategies": [
            {
              "type": "continue",
              "instructions": "The search tool is temporarily unavailable. Inform the user and provide your best answer from your knowledge."
            }
          ],
          "display_error_in_stage": true
        }
      }
    ]
  }
}
```

## 4. DIAL deployment as tool (with fallback handling)

Uses another DIAL application (a summarizer) as a tool. Demonstrates `DialDeploymentTool` with a custom system prompt for the tool call, content propagation, and fallback handling.

```json
{
  "applicationTypeSchemaId": "https://mydial.epam.com/custom_application_schemas/quickapps2",
  "applicationProperties": {
    "orchestrator": {
      "deployment": {
        "name": "gpt-4o"
      },
      "system_prompt": {
        "type": "custom",
        "variables": {},
        "content": "You are a research assistant. For long documents or articles that users share, use the summarize tool to create a concise summary before analyzing the content."
      }
    },
    "contexts": [],
    "tool_sets": [
      {
        "type": "dial-deployment",
        "name": "analysis-tools",
        "tools": [
          {
            "type": "deployment-tool",
            "open_ai_tool": {
              "type": "function",
              "function": {
                "name": "summarize",
                "description": "Summarize a long piece of text into a concise bullet-point summary.",
                "parameters": {
                  "type": "object",
                  "properties": {
                    "text": {
                      "type": "string",
                      "description": "The text to summarize."
                    },
                    "max_points": {
                      "type": "integer",
                      "description": "Maximum number of bullet points in the summary."
                    }
                  },
                  "required": ["text"]
                }
              }
            },
            "deployment": {
              "name": "claude-3-5-haiku",
              "parameters": {
                "temperature": 0.3,
                "max_tokens": 1024
              }
            },
            "system_prompt": {
              "type": "custom",
              "variables": {},
              "content": "You are a summarization specialist. Produce concise, accurate bullet-point summaries. Focus on key facts and decisions. Omit filler and repetition."
            },
            "content_propagation": {
              "propagate_history": false
            },
            "fallback_configuration": {
              "strategies": [
                {
                  "type": "continue",
                  "instructions": "Summarization failed. Please summarize the content yourself based on what the user shared."
                }
              ],
              "display_error_in_stage": false
            },
            "display": {
              "stage": {
                "name": "Summarizing document",
                "show": true
              }
            }
          }
        ]
      }
    ]
  }
}
```

## 5. Combined: REST API + MCP + DIAL deployment in one app

A "Research Assistant" that combines three tool set types in a single Quick App 2.0.

```json
{
  "applicationTypeSchemaId": "https://mydial.epam.com/custom_application_schemas/quickapps2",
  "applicationProperties": {
    "orchestrator": {
      "deployment": {
        "name": "gpt-4o",
        "parameters": {
          "temperature": 0.5,
          "max_iterations": 20
        }
      },
      "system_prompt": {
        "type": "custom",
        "variables": {},
        "content": "You are a research assistant. You have access to:\n- weather: look up current weather conditions\n- web_search: search the web for current information\n- summarize: summarize long texts\n\nUse these tools proactively when they would help answer the user's question. Combine results from multiple tools when appropriate."
      },
      "max_iterations": 20
    },
    "contexts": [],
    "tool_sets": [
      {
        "type": "rest-api",
        "name": "weather-api",
        "authorization": null,
        "tools": [
          {
            "type": "restapi-tool",
            "rest_api_method_info": {
              "method_url": "https://api.open-meteo.com/v1/forecast",
              "method_type": "get"
            },
            "open_ai_tool": {
              "type": "function",
              "function": {
                "name": "weather",
                "description": "Get current weather for a latitude/longitude coordinate.",
                "parameters": {
                  "type": "object",
                  "properties": {
                    "latitude": {
                      "type": "number",
                      "description": "Latitude.",
                      "parameter_info": { "type": "query", "key": "latitude" }
                    },
                    "longitude": {
                      "type": "number",
                      "description": "Longitude.",
                      "parameter_info": { "type": "query", "key": "longitude" }
                    },
                    "current": {
                      "type": null,
                      "const": "temperature_2m,wind_speed_10m,weather_code",
                      "parameter_info": { "type": "query", "key": "current" }
                    }
                  },
                  "required": ["latitude", "longitude", "current"]
                }
              }
            },
            "display": {
              "stage": {
                "name": "Fetching weather data",
                "show": true
              }
            }
          }
        ]
      },
      {
        "type": "mcp",
        "name": "search-tools",
        "mcp_server_info": {
          "url": "https://mcp.example.com/search",
          "protocol": "streamable_http",
          "authorization": {
            "type": "api_key",
            "key": "your-api-key",
            "name": "X-API-Key"
          }
        },
        "allowed_tools": ["web_search"],
        "fallback_configuration": {
          "strategies": [
            {
              "type": "continue",
              "instructions": "Web search is unavailable. Use your existing knowledge to answer."
            }
          ],
          "display_error_in_stage": false
        }
      },
      {
        "type": "dial-deployment",
        "name": "text-tools",
        "tools": [
          {
            "type": "dial-deployment-simple",
            "deployment_id": "claude-3-5-haiku"
          }
        ]
      }
    ],
    "starters": [
      "What's the weather like in Berlin right now?",
      "Search for the latest news about AI regulation",
      "Summarize this article for me: [paste URL or text]"
    ]
  }
}
```
